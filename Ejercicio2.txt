#include <iostream>
#include <sstream>
using namespace std;

int main() {
    string eco;
    cout << "Ingresa una frase: ";
    getline(cin, eco);
    istringstream iss(eco);
    string word;
    while (iss >> word) {
        cout << word << endl;
    }
    return 0;
}
